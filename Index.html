import java.util.*; public class Main { public static void main(String[] args) { Scanner sc = new Scanner(System.in); System.out.print("Enter Boy's Name: "); String boy = sc.nextLine().toLowerCase().replaceAll("\\s+", ""); System.out.print("Enter Girl's Name: "); String girl = sc.nextLine().toLowerCase().replaceAll("\\s+", ""); StringBuilder boyName = new StringBuilder(boy); StringBuilder girlName = new StringBuilder(girl); System.out.println("\n----- FLAMES Calculation Steps -----"); // Remove common letters for (int i = 0; i < boyName.length(); i++) { char ch = boyName.charAt(i); int index = girlName.indexOf(String.valueOf(ch)); if (ch != '*' && index != -1) { System.out.println("Removing common letter: " + ch); boyName.setCharAt(i, '*'); girlName.setCharAt(index, '*'); } } // Count remaining letters int count = 0; System.out.print("\nRemaining Boy Letters: "); for (int i = 0; i < boyName.length(); i++) { if (boyName.charAt(i) != '*') { System.out.print(boyName.charAt(i)); count++; } } System.out.print("\nRemaining Girl Letters: "); for (int i = 0; i < girlName.length(); i++) { if (girlName.charAt(i) != '*') { System.out.print(girlName.charAt(i)); count++; } } System.out.println("\n\nRemaining Count = " + count); ArrayList flames = new ArrayList<>(); flames.add('F'); flames.add('L'); flames.add('A'); flames.add('M'); flames.add('E'); flames.add('S'); int index = 0; System.out.println("\nFLAMES Elimination:"); while (flames.size() > 1) { index = (index + count - 1) % flames.size(); System.out.println("Removing: " + flames.get(index)); flames.remove(index); } char result = flames.get(0); System.out.println("\n----- RESULT -----"); switch (result) { case 'F': System.out.println("F = Friends"); break; case 'L': System.out.println("L = Lovers"); break; case 'A': System.out.println("A = Affection"); break; case 'M': System.out.println("M = Marriage"); break; case 'E': System.out.println("E = Enemies"); break; case 'S': System.out.println("S = Siblings"); break; } sc.close(); } }
